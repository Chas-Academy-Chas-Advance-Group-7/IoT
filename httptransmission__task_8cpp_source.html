100void httpTransmissionTask(void *pvParameters)
101{
102 safePrintf("[HTTP] Transmission task started.\n");
103
104 esp_log_level_set("ssl_client", ESP_LOG_DEBUG);
105
106 processed_data_t transferData;
107
108 static WiFiClientSecure client;
109 client.setCACert(digicertG2);
110 static HTTPClient http;
111
112 safePrintf("[HTTP] Task stack watermark: %u bytes\n",
113 uxTaskGetStackHighWaterMark(NULL)); // Debug stack usage
114
115 while (true)
116 {
117 // Wait for Wi-Fi connection
118 EventBits_t bits = xEventGroupWaitBits(networkEventGroup, NETWORK_CONNECTED_BIT, pdFALSE,
119 pdTRUE, portMAX_DELAY);
120 if ((bits & NETWORK_CONNECTED_BIT) == 0)
121 {
122 safePrintf("[HTTP] Waiting for Wi-Fi connection...\n");
123 vTaskDelay(pdMS_TO_TICKS(5000));
124 continue;
125 }
126
127 // Wait indefinitely for data
128 if (xQueueReceive(networkQueue, &transferData, portMAX_DELAY) == pdTRUE)
129 {
130 safePrintf("[HTTP] Preparing to send data (size %u bytes)...\n",
131 strlen(transferData.json));
132
133 bool success = false;
134
135 for (int attempt = 1; attempt <= MAX_HTTP_RETRIES; ++attempt)
136 {
137 safePrintf("[HTTP] Attempt %d to send data...\n", attempt);
138 safePrintf("[HTTP] Free heap: %u bytes\n", ESP.getFreeHeap());
139
140 http.begin(client, BACKEND_URL);
141 http.addHeader("Content-Type", "application/json");
142 // http.addHeader("X-API-Key", "Test_key");
143 http.addHeader("User-Agent", "ESP32Client/1.0");
144
145 int httpResponseCode = http.POST(String(transferData.json));
146
147 if (httpResponseCode > 0)
148 {
149 safePrintf("[HTTP] POST successful. Response code: %d\n", httpResponseCode);
150 String response = http.getString();
151 safePrintf("[HTTP] Server response: %s\n", response.c_str());
152 success = true;
153 http.end();
154 break;
155 }
156 else
157 {
158 safePrintf("[HTTP] Attempt %d failed. Error code: %d\n", attempt,
159 httpResponseCode);
160 http.end();
161
162 if (attempt < MAX_HTTP_RETRIES)
163 {
164 safePrintf("[HTTP] Retrying in %d ms...\n", RETRY_DELAY_MS);
165 vTaskDelay(pdMS_TO_TICKS(RETRY_DELAY_MS));
166 }
167 }
168 }
169
170 if (!success)
171 {
172 safePrintf("[HTTP] Transmission failed after %d attempts. Re-queueing data.\n",
173 MAX_HTTP_RETRIES);
174
175 if (xQueueSendToFront(networkQueue, &transferData, pdMS_TO_TICKS(100)) != pdTRUE)
176 safePrintf("[HTTP] Failed to re-queue data. Message may be lost.\n");
177 }
178 }
179 }
180}
