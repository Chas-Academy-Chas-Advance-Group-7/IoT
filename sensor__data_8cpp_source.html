144bool validateSensorData(const sensor_message_t *data)
145{
146 if (!data)
147 return false;
148
149 switch (data->sensor_type)
150 {
151 case SENSOR_TYPE_TEMPERATURE:
152 return (data->value >= -40.0f && data->value <= 85.0f);
153 case SENSOR_TYPE_HUMIDITY:
154 return (data->value >= 0.0f && data->value <= 100.0f);
155 default:
156 return false;
157 }
158}
