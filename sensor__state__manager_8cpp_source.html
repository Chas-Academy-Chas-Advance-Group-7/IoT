181void state_TransferPacketBatch()
182{
183 static int failed_transmission_attempts_counter = 0;
184 static unsigned long lastTransferTime = 0;
185 const unsigned long TRANSFER_INTERVAL = 150; // ms between packets
186
187 // Reset counters if recovering from an error
188 static bool recovering_from_error = false;
189
190 if (recovering_from_error)
191 {
192 failed_transmission_attempts_counter = 0;
193 lastTransferTime = millis();
194 recovering_from_error = false;
195 }
196
197 // Only attempt send if BLE central is connected
198 if (isCentralConnected() && queue_count > 0 && millis() - lastTransferTime >= TRANSFER_INTERVAL)
199 {
200 SensorPacket packet;
201
202 if (peekPacketFromBuffer(packet))
203 {
204 BLECharacteristic &charRef = getSensorCharacteristic();
205 uint8_t buffer[sizeof(SensorPacket)];
206 memcpy(buffer, &packet, sizeof(SensorPacket));
207
208 if (charRef.subscribed())
209 {
210 bool success = charRef.writeValue(buffer, sizeof(SensorPacket));
211
212 if (success)
213 {
214 // Commit removal and reset fail counter
215 commitPacketRemoval();
216 failed_transmission_attempts_counter = 0;
217
218 Serial.print("Sent packet to central: ");
219 Serial.print("Sensor ID: ");
220 Serial.print(packet.sensor_id);
221 Serial.print(", Temp: ");
222 Serial.print(packet.temperature);
223 Serial.print(", Hum: ");
224 Serial.println(packet.humidity);
225 }
226 else
227 {
228 failed_transmission_attempts_counter++;
229
230 Serial.print("Failed BLE write (attempt ");
231 Serial.print(failed_transmission_attempts_counter);
232 Serial.println(")");
233
234 // Escalate to ERROR_STATE if repeated failures exceed threshold
235 if (failed_transmission_attempts_counter >= MAX_FAILED_ATTEMPTS)
236 {
237 Serial.println("ERROR: BLE transmission failed too many times!");
238 recovering_from_error = true;
239 transitionToErrorState();
240 return; // Stop further processing
241 }
242 }
243 }
244 else
245 {
246 // Temporary condition; just log and retry next loop
247 Serial.println("Central not subscribed — skipping send.");
248 }
249
250 lastTransferTime = millis();
251 }
252 }
253
254 // If nothing in the buffer, move to IDLE (but only if no error)
255 if (queue_count == 0 && current_sensor_state != sensor_state::ERROR_STATE)
256 {
257 current_sensor_state = sensor_state::IDLE;
258 }
259}

277void state_BLEManagement()
278{
279 static int bleFailCounter = 0;
280 const int MAX_BLE_FAILS = 5;
281
282 // --- Added flag to prevent repeated advertising ---
283 static bool isAdvertising = false;
284
285 // Initialize BLE once
286 if (!bleInitialized)
287 {
288 if (setupBluetooth())
289 {
290 Serial.println("BLE initialized successfully.");
291 bleInitialized = true;
292 bleFailCounter = 0;
293
294 // Move to IDLE after successful init
295 current_sensor_state = sensor_state::IDLE;
296 }
297 else
298 {
299 bleFailCounter++;
300 Serial.print("BLE initialization failed. Attempt ");
301 Serial.println(bleFailCounter);
302
303 if (bleFailCounter >= MAX_BLE_FAILS)
304 {
305 Serial.println("Too many BLE init failures! Escalating to ERROR_STATE.");
306 transitionToErrorState();
307 }
308 return; // Skip further processing until successful init
309 }
310 }
311
312 // Reconnection logic
313 if (!isCentralConnected())
314 {
315 // --- Only re-advertise once per disconnect ---
316 if (!isAdvertising)
317 {
318 Serial.println("Central disconnected. Re-advertising BLE service...");
319 if (!BLE.advertise())
320 {
321 bleFailCounter++;
322 Serial.print("BLE advertise failed. Attempt ");
323 Serial.println(bleFailCounter);
324
325 if (bleFailCounter >= MAX_BLE_FAILS)
326 {
327 Serial.println("Too many BLE advertise failures! Escalating to ERROR_STATE.");
328 transitionToErrorState();
329 }
330 }
331 else
332 {
333 // Successfully started advertising, reset fail counter
334 Serial.println("BLE advertising restarted successfully.");
335 bleFailCounter = 0;
336 isAdvertising = true; // <-- prevent repeated spam
337 }
338 }
339 }
340 else
341 {
342 // Central is connected; reset fail counter and optionally move to IDLE
343 bleFailCounter = 0;
344
345 // --- Reset flag when central connects ---
346 if (isAdvertising)
347 {
348 isAdvertising = false;
349 }
350
351 if (current_sensor_state == sensor_state::BLE_MANAGEMENT)
352 {
353 current_sensor_state = sensor_state::IDLE;
354 Serial.println("Central connected. Returning to IDLE state.");
355 }
356 }

373void state_ErrorState()

376
377 // Log previous state
378 switch (previous_state_to_error_state)

380 case sensor_state::IDLE:

383

387 break;
388
389 case sensor_state::TRANSFER_PACKET_BATCH:
390 Serial.println(
391 "From TRANSFER_PACKET_BATCH: Issue occurred while transferring packet batch.");
392 break;
393
394 case sensor_state::BLE_MANAGEMENT:
395 Serial.println("From BLE_MANAGEMENT: Issue occurred while managing BLE.");
396 break;
397
398 case sensor_state::READ_FLASH_MEMORY:
399 Serial.println("From READ_FLASH_MEMORY: Issue occurred while reading flash memory.");
400 break;
401
402 default:
403 Serial.println("From unknown state.");
404 break;
405 }
406
407 // Reset state to IDLE
408 current_sensor_state = sensor_state::IDLE;
409 previous_state_to_error_state = sensor_state::IDLE;
410
411 Serial.println("System recovered: back to IDLE state.");
412}
